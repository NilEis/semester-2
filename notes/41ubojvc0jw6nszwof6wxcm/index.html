Implementierung des Betretens und Verlassen eines kritischen Bereichs

C

bool test_and_set(bool* target)
{
    bool res = *target;
    *target = true;
    return res;
}

Assembler

enter_region:
    tsl register, lock
    cmp register, #0
    jnz enter_region

    ret

leave_region:
    mov lock, #0
    ret
